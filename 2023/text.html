Modelado con SVM

Código 
set.seed(42)

objects_split <- initial_split(data, strata = feeling, prop = 0.7)
objects_training <- training(objects_split)
objects_testing <- testing(objects_split)
Código 
model_recipe <- recipe(feeling ~ news, data = objects_training) %>%
  step_tokenize(news) %>%
  step_stopwords(news) %>%
  step_tfidf(news) 

#prep(model_recipe) %>% bake(new_data = NULL) %>% glimpse()
Código 
#Naive Bayes
set.seed(234)
model_NB <- naive_Bayes(smoothness = .8) |> 
            set_mode("classification") |>  
            set_engine("naivebayes")
#SVM
set.seed(234)
model_SVM <- svm_linear() %>%
  set_mode("classification") %>%
  set_engine("LiblineaR")
Código 
we_svm <- workflow() |> 
          add_recipe(model_recipe) |> 
          add_model(model_SVM)

## Modelado con SVM

```{r}
set.seed(42)

objects_split <- initial_split(data, strata = feeling, prop = 0.7)
objects_training <- training(objects_split)
objects_testing <- testing(objects_split)
```


```{r}
model_recipe <- recipe(feeling ~ news, data = objects_training) %>%
  step_tokenize(news) %>%
  step_stopwords(news) %>%
  step_tfidf(news) 

#prep(model_recipe) %>% bake(new_data = NULL) %>% glimpse()
```

```{r}
#Naive Bayes
set.seed(234)

set.seed(234)
model_SVM <- svm_linear() %>%
  set_mode("classification") %>%
  set_engine("LiblineaR")
```


```{r}
we_svm <- workflow() |> 
          add_recipe(model_recipe) |>
```
